# 防抖

你尽管去触发事件，但是我一定在事件触发n秒后才执行，如果你在一个事件触发的n秒内又触发了这个事件， 则我就以新的事件的时间为准，n秒后才执行。总之，就是要等你触发完事件n秒内不再触发事件，我才执行。

  • 持续触发不执行
  • 不触发的一段时间之后再执行
const debounce = function(func, delay) {
    let timer;
    return function () {
        // 如果持续触发，就清除定时器，定时器的回调就不会执行。
        clearTimeout(timer);
        timer = setTimeout(() => {
            func.apply(this, arguments)
        }, delay)
    }
}

# 新增立刻执行需求

不希望非要等到事件停止触发后才执行，希望立即执行函数，然后等到停止触发n秒后，才可以重新触发执行。

// 加个immediate参数判断是否是立刻执行
const debounce = function(func, delay, immediate) {
    let timer;
    return function () {
        const context = this;
        const args = arguments;
        if (timer) clearTimeout(timer); // 执行clearTimeout方法并不会将timer置为null
        if (immediate) {
            // 若已经执行过，不再执行。
            const callNow = !timer;
            timer = setTimeout(() => {
                timer = null;
            }, delay);
            if (callNow) func.apply(context, args);
        } else {
            timer = setTimeout(() => {
                func.apply(context, args);
            }, delay)
        }
    }
}

# 节流

节流意味让函数有节制地执行，而不是毫无节制的触发一次就执行一次。即在一段时间内，只执行一次。步骤分解如下：

  • 持续触发并不会执行多次
  • 到一定时间再去执行

节流原理：持续触发事件，每隔一段时间只执行一次事件。

const throttle = function(func, delay) {
    let run = true;
    return function () {
        if (!run) {
            return;    
        }
        run = false;
        setTimeout(() => {
            func.apply(this, arguments);
            run = true;
        }, delay)
    }
}

# 使用时间戳

使用时间戳，当触发事件时取出当前的时间戳（now），然后减去之前的时间戳（prevTime）。若大于设置的时间周期， 就执行函数，然后更新时间戳为当前的时间戳（prevTime = now）；若小于就不执行。

const throttle = function(func, delay) {
    let prevTime = +new Date();
    return function() {
        const context = this;
        const args = arguments;
        const now = +new Date();
        if (now - prevTime > delay) {
            func.apply(context, args);
            prevTime = now;
        }
    }
}

# 使用计时器

触发事件时，设置一个定时器。再次触发事件的时候，若定时器存在就不执行；直到定时器执行，然后执行函数，清空定时器，同时设置下个定时器。

const throttle = function(func, delay) {
    let timer;
    return function() {
        const context = this;
        const args = arguments;
        if (!timer) {
            timer = setTimeout(() => {
                timer = null;
                func.apply(context, args);
            }, delay)
        } 
    }
}

# 两种方法缺点（特点）

  • 时间戳方法事件会立刻执行，计时器方法事件会在n秒后第一次执行
  • 时间戳方法事件停止触发后没有办法再执行事件，计时器方法事件停止触发后依然会再执行一次事件

# 双剑合璧

需求：鼠标移入能立刻执行，停止触发的时候还能再执行一次。

const throttle = function(func, delay) {
    let timer, context, args,
        previous = 0;
    
    const later = function() {
        previous = +new Date(); // 更新previous值，用于更新计算remaining的值
        timer = null; // 需要重置timer为null，否则当后续触发事件时走不到else if此分支
        func.apply(context, args);
    }
    const throttled = function() {
        console.log('previous', previous);
        const now = +new Date();
        // 下次触发func剩余的时间
        const remaining = delay - (now - previous);
        context = this;
        args = arguments;
        // 若没有剩余时间或者修改了系统时间
        if (remaining <= 0 || remaining > delay) {
            console.log('首次移入立即执行');
            if (timer) {
                clearTimeout(timer);
                timer = null;
            }
            previous = now;
            func.apply(context, args);
        } else if (!timer) {
            console.log('2');
            timer = setTimeout(later, remaining);
        }
    }
    return throttled;
}
