/*
*   链表相交
*   @desc：关键在于如何判断两个链表是否相交
*/
const getIntersectionNode = function(headA, headB) {
    if (!headA || !headB) return null;
    let curA = headA,
        curB = headB;
    while (curA || curB) {
        if (curA === curB) return curA;
        curA = curA ? curA.next : headB;
        curB = curB ? curB.next : headA;
    }
    return null;
};

/*
*   返回倒数第 k 个节点
*   @desc 1. 先计算出整个链表的长度 len；
*         2. 用链表长度 - k 得到该节点的正数位置
*         3. 遍历链表即可
*/
const kthToLast = function (head, k) {
    if (!head) return null;
    let len = 0,
        list = head;
    while (list) {
        list = list.next;
        len++;
    }
    len = len - k;
    while (len) {
        head = head.next;
        len--;
    }
    return head.val;
};

/*
*   两数相加/链表求和
*   l1：7->3->7  l2：4->9->6
*   输出结果：1->3->4->1（特殊点在于若链表的最后一个节点相加仍然大于10，则需要进一位就是多加一个链表节点）
*/
const addTwoNumbers = function (l1, l2) {
    let sum = 0,
        s = new ListNode(null),
        head = s,
        remainder = 0;
    while (l1 || l2 || remainder) {
        // 注意判断 l1 或者 l2 为null的情况
        sum = (l1 && l1.val) + (l2 &&l2.val) + remainder;
        remainder = 0;
        if (sum >= 10) {
            sum = sum - 10;
            remainder = 1;
        }
        const node = new ListNode(sum);
        s.next = node;
        s = node;
        l1 = l1 && l1.next;
        l2 = l2 && l2.next;
    }
    return head.next;
};

/*
*   回文链表
*   @desc 以链表 1->2->3->2->1 为例
*   1. 通过快慢指针求出中间节点,同时一边反转中间节点前部分。遍历完后：fast=1，slow=3，preHead=2->1->null
*   2. 对奇数链表做特殊处理，slow = slow.next;
*   3. 遍历后半部分节点，依次与反转后的前半部分节点比较。
*   需要注意：链表为空或者链表只有头结点时，也属于回文。
*/
const isPalindrome = function (head) {
    // if (!head || !head.next) return false;
    let prev = head,
        leftList = null,
        fast = head,
        slow = head;
    while (fast && fast.next) {
        prev = slow;
        fast = fast.next.next;
        slow = slow.next;
        prev.next = leftList;
        leftList = prev;
    }
    // 链表为奇数时，slow需要指向下一个指针
    if (fast) {
        slow = slow.next;
    }
    while (slow && leftList) {
        if (slow.val !== leftList.val) return false;
        slow = slow.next;
        leftList = leftList.next;
    }
    return true;
};

/*
*   删除排序链表中的重复元素（已排好序的链表）
*   @desc https://leetcode-cn.com/problems/remove-duplicates-from-sorted-list/solution/shan-chu-pai-xu-lian-biao-zhong-de-zhong-fu-yuan-s/
*/
const deleteDuplicates = function (head) {
    let cur = head;
    while (cur && cur.next) {
        if (cur.next.val === cur.val) {
            cur.next = cur.next.next;
        } else {
            cur = cur.next;
        }
    }
    return head;
};

const mergeTwoLists = function (l1, l2) {
    if (!l1) return l2;
    if (!l2) return l1;
    if (l1.val > l2.val) {
        l2.next = mergeTwoLists(l1, l2.next);
        return l2;
    } else {
        l1.next = mergeTwoLists(l1.next, l2);
        return l1;
    }
};

/*
*   反转链表：双指针
*   @desc 1. 定义两个指针：prev和cur；prev在前cur在后。
*   2. 每次让prev的next指向cur，实现一次局部反转。
*   3. 局部反转完成后，prev和cur同时往前移动一个位置。
*   4. 循环上述步骤，直至prev到达链表尾部。
*/
const reverseList = function (head) {
    if (!head || !head.next) return head;
    let cur = null,
        prev = head;
    while (prev) {
        let t = prev.next;
        prev.next = cur;
        cur = prev;
        prev = t;
    }
    return cur;
};

/*
*   反转链表：递归
*   题解：https://leetcode-cn.com/problems/fan-zhuan-lian-biao-lcof/solution/fan-zhuan-lian-biao-yi-dong-de-shuang-zhi-zhen-jia/
*   @desc 1. 使用递归函数，一直递归到链表的最后一个结点，该结点就是反转后的头结点，记作 ret。
*   2. 此后每次函数在返回的过程中，让当前结点的下一个结点的 next 指针指向当前节点。
*   3. 同时让当前结点的 next 指针指向 null，从而实现从链表尾部开始的局部反转。
*   4. 当递归函数全部出栈后，链表反转完成。
*/
const reverseList = function (head) {
    if (!head || !head.next) return head;
    let ret = reverseList(head.next);
    head.next.next = head;
    head.next = null;
    return ret;
};

/*
*   链表的中间结点（带有头结点 head 的非空单链表）
*   @desc：双指针/快慢指针 求中间结点
*/
const middleNode = function (head) {
    let fast = head,
        slow = head;
    while (fast && fast.next) {
        fast = fast.next.next;
        slow = slow.next;
    }
    return slow;
};

/*
*   移除链表元素
*   @desc：通过哨兵节点
*   当要删除的一个或多个节点位于链表的头部时。
*/
const removeElements = function (head, val) {
    let s = new ListNode(null),
        cur = head;
    s.next = head;
    let prev = s;
    while (cur) {
        if (cur.val === val) {
            prev.next = cur.next
        } else {
            prev = cur;
        }
        cur = cur.next;
    }
    return s.next;
};

/*
*   环形链表（判断链表中是否有环）
*   @desc：快慢指针，链表中若存在环，则快慢指针终会相遇
*/
const hasCycle = function (head) {
    if (!head || !head.next) return false;
    let fast = head,
        slow = head;
    while (fast && fast.next) {
        fast = fast.next.next;
        slow = slow.next;
        if (fast === slow) return true;
    }
    return false;
};
